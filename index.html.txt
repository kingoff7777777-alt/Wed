function setLang(lang) {
  document.getElementById("desc").innerText = data[lang].desc;
  document.getElementById("friendTitle").innerText = data[lang].friendTitle;
  document.getElementById("friend").innerText = data[lang].friend;
  document.getElementById("copy").innerText = data[lang].copy;
}

function toggleMusic() {
  const music = document.getElementById("music");
  const btn = document.getElementById("musicBtn");

  if (music.paused) {
    music.play();
    btn.innerText = "⏸ Pause Music";
  } else {
    music.pause();
    btn.innerText = "🎧 Play Music";
  }
}

document.getElementById("intro").addEventListener("click", () => {
  document.getElementById("intro").style.display = "none";
  document.getElementById("main").classList.remove("hidden");
  document.getElementById("music").play();
  setLang("vi");
});
